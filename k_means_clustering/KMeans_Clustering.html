for i in range(2001,2018):
    comunCrimes = list()    
    print('Reading file '+str(i)+'...')
    fileName = 'day_type_description_crap_'+str(i)+'.csv'
    df = pd.read_csv(fileName)
    del df['Unnamed: 0']       #Deleting first column
    del df['YearMonthDay']     #Deleting second column

    df['CommunityArea'] = df['CommunityArea'].astype(int)
